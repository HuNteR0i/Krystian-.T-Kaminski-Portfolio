void lvl_3() {

    int rand_num = rand() % 51;
    data().guess = 15;

    while (data().guess != 0) {

        system("cls");

        cout << "lvl.3" << endl;
        cout << data().line << endl;

        for (int i = 0; i < 51; i++){

            if (data().new_row == i)
            {
                cout << " " << i << " " << endl;
            }
            else if (i == data().curent_pos) {
                cout << "<" << i << ">";
            }
            else {
                cout << " " << i << " ";
            }
            
            
        }

        cout << endl << data().line << endl;

        cout << "YOU GOT " << data().guess << " GUESS";

        data().key = _getch();

        if (data().key == 224) {
            data().key = _getch();

            switch (data().key) {
            case 75:
                data().curent_pos--;
                if (data().curent_pos < 0) data().curent_pos = 50;
                break;
            case 77:
                data().curent_pos++;
                if (data().curent_pos > 50) data().curent_pos = 0;
                break;
            }
        }
        else if (data().key == 13) {

            if (data().curent_pos != rand_num) {
                data().guess--;
                continue;
            }

            else if (data().curent_pos == rand_num) {
                break;
            }

            break;
        }
    }

    system("cls");

    cout << endl << endl;

    if (data().curent_pos != rand_num) {
        cout << "BAD LUCK MATE" << endl;
        data().lose++;

    }
    else if (data().curent_pos == rand_num) {
        cout << "GOOD MATE" << endl;
        data().win++;
    }

    cout << data().line << endl;

}
