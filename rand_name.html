# function for menu
def print_menu():
    position = 0
    print("  EXIT ")
    while True:

        key = msvcrt.getch()

        # find key input
        if key == b'\xe0':
            key2 = msvcrt.getch()

            match key2:
                case b'M': # right arrow
                    position += 1
                    clear()
                case b'K': # left arrow
                    position -= 1
                    clear()
                case _: # invalid option
                    clear()
                    print("invalid option")
                    time.sleep(1)
        # if position overflows
        if position < 0:
            position = 1
        elif position > 1:
            position = 0

        # print menu with current position
        match position:
            case 0:
                print ("  EXIT ")
            case 1:
                print ("  Generate  ")
    
        # if ENTER is pressed
        if key == b'\r':

            # send to funktion names()
            if position == 0:
                clear()
                names()
            # exit program
            elif position == 1:
                clear()
                sys.exit()
 

# start of the program
print_menu()
