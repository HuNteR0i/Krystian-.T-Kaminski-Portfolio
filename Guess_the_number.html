void game() {
    int level, rand_num, guess, tries;

    

    switch (level) {
    case 1: {
        cout << "The game has begun!" << endl;
        tries = 1;
        rand_num = rand() % 5 + 1;
        while (tries > 0) {
            cout << "Guess a number between 1 and 5: ";
            cin >> guess;

            if (guess == rand_num) {
                cout << "You win!" << endl;
                break;
            }
            else {
                tries--;
                cout << "Wrong, try again." << endl;
                cout << "You have " << tries << " tries left." << endl;
            }
        }
        if (tries == 0) {
            cout << "You lose. The number was " << rand_num << "." << endl;
        }
        break;
    }
    case 2: {
        cout << "The game has begun!" << endl;
        tries = 3;
        rand_num = rand() % 10 + 1;
        while (tries > 0) {
            cout << "Guess a number between 1 and 10: ";
            cin >> guess;
            if (guess == rand_num) {
                cout << "You win!" << endl;
                break;
            }
            else {
                tries--;
                cout << "Wrong, try again." << endl;
                cout << "You have " << tries << " tries left." << endl;
                if (guess > rand_num)
                    cout << "The number is lower." << endl;
                else
                    cout << "The number is higher." << endl;
            }
        }
        if (tries == 0) {
            cout << "You lose. The number was " << rand_num << "." << endl;
        }
        break;
    }
    case 3: {
        cout << "The game has begun!" << endl;
        tries = 5;
        rand_num = rand() % 20 + 1;
        while (tries > 0) {
            cout << "Guess a number between 1 and 20: ";
            cin >> guess;
            if (guess == rand_num) {
                cout << "You win!" << endl;
                break;
            }
            else {
                tries--;
                cout << "Wrong, try again." << endl;
                cout << "You have " << tries << " tries left." << endl;
                if (guess > rand_num)
                    cout << "The number is lower." << endl;
                else
                    cout << "The number is higher." << endl;
            }
        }
        if (tries == 0) {
            cout << "You lose. The number was " << rand_num << "." << endl;
        }
        break;
    }

    default:
        cout << "Invalid difficulty level!" << endl;
        break;
    }
}

void beginning() {

        }

}
